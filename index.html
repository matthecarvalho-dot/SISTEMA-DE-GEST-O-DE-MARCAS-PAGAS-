function init() { carregarNomeAdmin(); carregarDados(); atualizarData(); setInterval(atualizarData, 60000); document.getElementById('formData').value = new Date().toISOString().split('T')[0]; document.getElementById('vendedoraSelect').value = 'todas'; filtrarPorVendedora(); // Auto-save a cada 30 segundos setInterval(salvarDados, 30000); window.addEventListener('beforeunload', salvarDados); // Fechar modais com Escape document.addEventListener('keydown', function(e) { if (e.ctrlKey && e.key === 'Enter') { e.preventDefault(); document.getElementById('formFechamento').requestSubmit(); } if (e.ctrlKey && e.key === 's') { e.preventDefault(); salvarDados(); showToast('Dados salvos! 💾', 'success'); } if (e.key === 'Escape') { fecharAdminModal(); fecharImportModal(); } }); }
